1) Create daleads Table:

DROP TABLE IF EXISTS public.daleads;

CREATE TABLE public.daleads (
  lead_Id SERIAL PRIMARY KEY,
  full_Name TEXT NOT NULL,
  email TEXT NOT NULL,
  phone TEXT,
  course_Interested TEXT,
  lead_source TEXT DEFAULT 'web_site',
  add_Source TEXT,
  createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  lead_status TEXT DEFAULT 'New',
  Remarks TEXT DEFAULT 'NA'
);



Sync with Neon DB:
npx prisma db pull
npx prisma generate

export async function GET(req) {

async function getAllLeads(req) {
  try {
    const leads = await fetchLeadsFromDB();
    return new Response(JSON.stringify({ ok: true, data: leads }), {
      status: 200,
    });
  } catch (error) {
    console.error("getAllLeads error:", error);
    return new Response(
      JSON.stringify({ ok: false, error: "Failed to fetch leads" }),
      { status: 500 }
    );
  }
}
